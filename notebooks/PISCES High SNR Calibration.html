import tools
from tools.wavecal import computeWavecal
import glob
from params import Params
import numpy as np
import logging as log
from tools.initLogger import initLogger
from astropy.io import fits as pyf
from tools.image import Image


par = Params()
initLogger(par.exportDir+'/IFSWavecalHighSNR.log')
from tools.locate_psflets import PSFLets
from tools.wavecal import gethires
lam = np.loadtxt(par.wavecalDir + "lamsol.dat")[:, 0]
allcoef = np.loadtxt(par.wavecalDir + "lamsol.dat")[:, 1:]

par.lamlist = [655, 665, 675, 685, 695, 705, 715, 725]
par.filelist = np.sort(glob.glob(par.wavecalDir + "*nm*.fits"))
print lam,lam.shape
imlist = []
print len(par.filelist)
for j in range(len(par.lamlist)):
    im = Image(filename=par.filelist[j])
    imlist += [im]
psftool = PSFLets()
psftool.geninterparray(lam, allcoef)

#psftool.genpixsol(lam, allcoef, lam1=lam1/1.04, lam2=lam2*1.03)
#psftool.savepixsol(outdir=outdir)
savehiresimages = True
hires_arrs = []
xindx = np.arange(-100, 101)
xindx, yindx = np.meshgrid(xindx, xindx)
for i in range(len(lam)):
    xpos, ypos = psftool.return_locations(lam[i], allcoef, xindx, yindx)
    xpos = np.reshape(xpos, -1)
    ypos = np.reshape(ypos, -1)
    print "Shape:",imlist[i].data.shape
    hiresarr = gethires(xpos, ypos, imlist[i],upsample=3,nsubarr=3)
    hires_arrs += [hiresarr]

    if savehiresimages:
        di, dj = hiresarr.shape[0], hiresarr.shape[2]
        outim = np.zeros((di*dj, di*dj))
        for ii in range(di):
            for jj in range(di):
                outim[ii*dj:(ii + 1)*dj, jj*dj:(jj + 1)*dj] = hiresarr[ii, jj]
        out = pyf.HDUList(pyf.PrimaryHDU(hiresarr.astype(np.float32)))
        out.writeto(par.wavecalDir + 'hires_psflets_lam%d.fits' % (lam[i]), clobber=True)
